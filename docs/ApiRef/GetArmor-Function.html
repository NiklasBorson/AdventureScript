function GetArmor($kind:ArmorKind) : Item
{
    switch ($kind)
    {
        case ArmorKind.Head
        {
            $return = $headArmor;
        }
        case ArmorKind.Torso
        {
            $return = $torsoArmor;
        }
        case ArmorKind.Leg
        {
            $return = $legArmor;
        }
    }
}
