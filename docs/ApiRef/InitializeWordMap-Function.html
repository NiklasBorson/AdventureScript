function InitializeWordMap()
{
    if ($isNowDark)
    {
        foreach (var $item) where Location == player
        {
            AddItemWords($item);
        }
    }
    else
    {
        foreach (var $item)
        {
            if (IsAccessible($item))
            {
                AddItemWords($item);
            }
        }
        foreach (var $dir:Direction)
        {
            var $door = GetLink(player.Location, $dir);
            if ($door.DoorState != DoorState.None)
            {
                AddAdjectives($"{$dir}", $door);
                AddItemWords($door);
            }
        }
    }
}
