Source

function SetLink($from:Item, $to:Item, $dir:Direction)
{
    switch ($dir)
    {
        case Direction.North
        {
            $from.LinkN = $to;
        }
        case Direction.South
        {
            $from.LinkS = $to;
        }
        case Direction.East
        {
            $from.LinkE = $to;
        }
        case Direction.West
        {
            $from.LinkW = $to;
        }
        case Direction.Up
        {
            $from.LinkU = $to;
        }
        case Direction.Down
        {
            $from.LinkD = $to;
        }
    }
}
