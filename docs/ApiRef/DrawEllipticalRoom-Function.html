Source

function DrawEllipticalRoom($room:Item, $width:Int, $height:Int) : Int
{
    BeginDrawing($width, $height);
    DrawEllipse(0, 0, $width, $height, -1, -16777216, 15);
    DrawDoors($room, 0, 0, $width, $height, -1);
    return EndDrawing();
}
