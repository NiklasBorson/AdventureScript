function TurnOnLight($item:Item)
{
    if ($item.LightState != LightState.On)
    {
        $item.LightState = LightState.On;
        Message($"The {Label($item)} is now on.");
        OnLightActivated($item);
    }
    else
    {
        Message($"The {Label($item)} is already on.");
    }
}
