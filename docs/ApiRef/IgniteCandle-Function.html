function IgniteCandle($item:Item)
{
    if ($item.LightState != LightState.Lit)
    {
        $item.LightState = LightState.Lit;
        Message($"The {Label($item)} is now lit.");
        OnLightActivated($item);
    }
    else
    {
        Message($"The {Label($item)} is already lit.");
    }
}
