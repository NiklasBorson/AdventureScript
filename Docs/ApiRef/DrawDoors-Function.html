function DrawDoors($room:Item, $left:Int, $top:Int, $width:Int, $height:Int, $backColor:Int)
{
    var $openW = false;
    var $openE = false;
    if ($room.LinkW != null && !$room.LinkW.IsHidden)
    {
        if ($room.LinkW.DoorState != DoorState.None)
        {
            DrawRectangle($left, $top + $height / 2 - 25, 17, 50, -1, -16777216, 2);
        }
        else
        {
            DrawRectangle($left, $top + 15, 17, $height - 30, $backColor, 0, 0);
            $openW = true;
        }
    }
    if ($room.LinkE != null && !$room.LinkE.IsHidden)
    {
        if ($room.LinkE.DoorState != DoorState.None)
        {
            DrawRectangle($left + $width - 17, $top + $height / 2 - 25, 17, 50, -1, -16777216, 2);
        }
        else
        {
            DrawRectangle($left + $width - 17, $top + 15, 17, $height - 30, $backColor, 0, 0);
            $openE = true;
        }
    }
    if ($room.LinkN != null && !$room.LinkN.IsHidden)
    {
        if ($room.LinkN.DoorState != DoorState.None)
        {
            DrawRectangle($left + $width / 2 - 25, $top, 50, 17, -1, -16777216, 2);
        }
        else
        {
            var $x1 = $openW ? 0 : 15;
            var $x2 = $openE ? $width : $width - 15;
            DrawRectangle($left + $x1, $top, $x2 - $x1, 17, $backColor, 0, 0);
        }
    }
    if ($room.LinkS != null && !$room.LinkS.IsHidden)
    {
        if ($room.LinkS.DoorState != DoorState.None)
        {
            DrawRectangle($left + $width / 2 - 25, $top + $height - 17, 50, 17, -1, -16777216, 2);
        }
        else
        {
            var $x1 = $openW ? 0 : 15;
            var $x2 = $openE ? $width : $width - 15;
            DrawRectangle($left + $x1, $top + $height - 17, $x2 - $x1, 17, $backColor, 0, 0);
        }
    }
    if ($room.LinkU != null && !$room.LinkU.IsHidden)
    {
        var $x = $left + $width / 2 - 40;
        var $y = $top + $height / 2 - 35;
        DrawRectangle($x, $y, 35, 70, 0, -16777216, 2);
        DrawRectangle($x, $y + 10, 35, 2, -16777216, 0, 0);
        DrawRectangle($x, $y + 20, 35, 2, -16777216, 0, 0);
        DrawRectangle($x, $y + 30, 35, 2, -16777216, 0, 0);
        DrawRectangle($x, $y + 40, 35, 2, -16777216, 0, 0);
        DrawRectangle($x, $y + 50, 35, 2, -16777216, 0, 0);
        DrawRectangle($x, $y + 60, 35, 2, -16777216, 0, 0);
    }
    if ($room.LinkD != null && !$room.LinkD.IsHidden)
    {
        var $x = $left + $width / 2 + 5;
        var $y = $top + $height / 2 - 35;
        DrawRectangle($x, $y, 35, 70, -16777216, 0, 0);
        DrawRectangle($x + 4, $y + 2, 27, 8, -1, 0, 0);
        DrawRectangle($x + 5, $y + 13, 25, 8, -1, 0, 0);
        DrawRectangle($x + 6, $y + 24, 23, 7, -1, 0, 0);
        DrawRectangle($x + 7, $y + 34, 21, 7, -1, 0, 0);
        DrawRectangle($x + 8, $y + 44, 19, 6, -1, 0, 0);
        DrawRectangle($x + 9, $y + 53, 17, 6, -1, 0, 0);
        DrawRectangle($x + 10, $y + 62, 15, 5, -1, 0, 0);
    }
}
