function DescribeTable($table:Item)
{
    DescribeCommon($table);
    var $haveItems = false;
    foreach (var $item) where Location == $table
    {
        if (!$haveItems)
        {
            Message($"On the {$table.Noun} are the following:");
            $haveItems = true;
        }
        Message($" - A {ItalicLabelWithState($item)}.");
    }
}
