IsClosedOrLocked Function

Returns true if the specified door state is either closed or locked.

function IsClosedOrLocked(
    $fromValue : DoorState
    ) : Bool;

Parameters

$fromValue

DoorState value.

Return value

Returns true for Closed or Locked.

Source

map IsClosedOrLocked DoorState -> Bool
{
    None -> false,
    Open -> false,
    Closed -> true,
    Locked -> true
}
