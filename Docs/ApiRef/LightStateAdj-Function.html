LightStateAdj Function

Converts a LightState value to an adjective.

function LightStateAdj(
    $fromValue : LightState
    ) : String;

Source

map LightStateAdj LightState -> String
{
    None -> "",
    Off -> "off",
    On -> "on",
    Unlit -> "unlit",
    Lit -> "lit"
}
